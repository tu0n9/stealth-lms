(function deepStealth() {
    console.log("%c[SYSTEM] Kích hoạt!", "color: #ff00ff; font-weight: bold;");

    // 1. Dọn dẹp
    delete document.visibilityState;
    delete document.hidden;

    // 2. chặn ở tầng prototype
    Object.defineProperty(Document.prototype, 'visibilityState', {
        get: function() { return 'visible'; },
        configurable: true
    });

    Object.defineProperty(Document.prototype, 'hidden', {
        get: function() { return false; },
        configurable: true
    });

    // 3. chặn hàm kiểm tra focus
    Document.prototype.hasFocus = function() { return true; };
    window.hasFocus = function() { return true; };

    // 4. Chặn ngầm sự kiện
    const block = (e) => e.stopImmediatePropagation();
    document.addEventListener('visibilitychange', block, true);
    window.addEventListener('blur', block, true);
    window.addEventListener('focusout', block, true);
    document.addEventListener('mouseleave', block, true);
})();

    const workerCode = `
        setInterval(() => {
            postMessage('tick');
        }, 16);
    `;
    const blob = new Blob([workerCode], { type: 'application/javascript' });
    const worker = new Worker(URL.createObjectURL(blob));

    // Lưu trữ các hàm RAF gốc, và tạo bộ lưu trữ callback
    const originalRaf = window.requestAnimationFrame;
    const originalCaf = window.cancelAnimationFrame;
    let rafId = 0;
    const rafCallbacks = new Map();

    // Mỗi khi Worker báo 'tick', chạy tất cả các callback đang đợi
    worker.onmessage = () => {
        const now = performance.now(); // Lấy mốc thời gian thực chuẩn xác
        
        // Clone danh sách ra để chạy, tránh bị loop vô tận nếu callback lại gọi thêm RAF
        const callbacksToRun = new Map(rafCallbacks);
        rafCallbacks.clear();
        
        callbacksToRun.forEach((callback) => {
            try {
                callback(now);
            } catch (error) {
                console.error("Lỗi trong rAF fake:", error);
            }
        });
    };

    // Ghi đè hàm rAF của trình duyệt
    window.requestAnimationFrame = function(callback) {
        rafId++;
        rafCallbacks.set(rafId, callback);
        return rafId;
    };

    // Ghi đè hàm hủy rAF
    window.cancelAnimationFrame = function(id) {
        rafCallbacks.delete(id);
    };
})();
